    \subsection*{Conception détaillée}

        
    \begin{function}
        \SetAlgoLined
        \caption{reconnaitreLesMots(uneChaine : Chaine de Caractère) : Ensemble$<$Mot$>$ }
        \Declaration{long,d,i : Naturel, lesMots : Ensemble$<$Mot$>$}
        \Deb{
        	d $\gets$ 1\;
        	\Tq{}{}
            \Si{reconnaitreUnMot(ch) $\ne$ NULL}{
                Ensemble.ajouter(lesMots, reconnaitreUnMot(ch))\;
            }
            \Retour{lesMots}
        }
    \end{function}
    
    \begin{function}
        \SetAlgoLined
        \caption{reconnaitreUnMot(uneChaine : Chaine de Caractère) : Mot}
        \Declaration{}
        \Deb{\Retour{creerUnMot(uneChaine)}}
    \end{function}
    
    \begin{function}
        \SetAlgoLined
        \caption{creerSousChaine(uneChaine : Chaine de Caractère, position : NaturelNonNul, long : Naturel ) : Chaine de Caractère}
        \Declaration{resultat : Chaine de caracteres, i : Naturel}
        \Deb{
        	resultat $\gets$ ""\;
        	\Pour{i $\gets$ 1 à long-1}{
        		resultat $\gets$ resultat + iemeCaractere(uneChaine,position+i) \;
        	}
            \Retour{resultat}
        }
    \end{function}
