    \begin{procedure}
        \SetAlgoLined
        \caption{insererLettre(\textbf{E/S :} unArbre: ArbreDeLettres, \textbf{E/S :} uneLettre: Caractere) }
        \Declaration{temp : ArbreDeLettres}
        \Deb{
            temp = A\_obtenirFilsDroit(unArbre) \\
            A\_fixerLettre(temp) $\leftarrow$ uneLettre \\
            A\_fixerFilsDroit(unArbre, temp)\\
        }
    \end{procedure}
    
    \begin{function}
        \SetAlgoLined
        \caption{estPresentDansFils(unArbre : ArbreDeLettres, uneLettre: Caractere) : Booleen}
        \Declaration{}
        \Deb{

        	\Retour{A\_obtenirLettre(unArbre) = uneLettre}
        }
    \end{function}

    \begin{procedure}
        \SetAlgoLined
        \caption{insererMot(\textbf{E/S :} unArbre: ArbreDeLettres, \textbf{E/S :} unMot : Mot) }
        \Declaration{temp : ArbreDeLettres}
        \Deb{
            \Si{estPresentDansFils(unArbre, M\_iemeCaractere(Mot, 1))}
            {
                temp $\leftarrow$ A\_obtenirFilsGauche(unArbre) \\
                unMot $\leftarrow$ supprimerIemeLettre(unMot, 1) \\
                insererMot(temp, unMot)
            }
            \Sinon
            {
                temp $\leftarrow$ A\_obtenirFilsDroit(unArbre)\\
                \Si{A\_estVide(temp)}
                {
                    A\_fixerFilsDroit(unArbre, M\_iemeCaractere(Mot, 1)) \\
                    temp $\leftarrow$ A\_obtenirFilsDroit(unArbre) \\
                    temp $\leftarrow$ A\_obtenirFilsGauche(temp) \\
                    insererMot
                    
                }
            }
            
        }
    \end{procedure}
    
    \begin{function}
        \SetAlgoLined
        \caption{genererArbreEnFichierTexte(fic : FichierTexte): ArbreDeLettres}
        \Declaration{lesMots : Ensemble$<$Mot$>$, unArbre : ArbreDeLettres}
        \Deb{
            lesMots $\leftarrow$ genererEnsembleDeMot(fic) \\
            unArbre $\leftarrow$ genererArbreAvecEnsembleDeMots(lesMots) \\
        	\Retour{}
        }
    \end{function}
